[section ram, code]
	store ret.value
	push 2
	store i_input2
	push 2
	store sum_input2
calculator:
	push label_0
	push i_input2
	wide_store
	push label_1
	push i_input2
	wide_add
	store label_2.type
	store label_2.value
	push label_2
	push sum_input2
	wide_store
	push label_3
	store ret.value
	jump print_value
label_3:
	push halt
	store ret.value
	push label_4
	push i_input2
	wide_div
	store label_5.type
	store label_5.value
	push label_5
	push sum_input2
	wide_store
	jmp_emp ret.value
print_value:
	write
	jmp_emp ret.value
	jump halt

    .value: dd 0x10 ; Offset for `value`

	hlt

    .value: dd 3 ; Offset for `value`
